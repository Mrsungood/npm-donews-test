# npm-donews-test

> A Vue.js project

## Install

``` shell
npm install npm-donews-test -S
```

## Quick Start
``` javascript
import Vue from 'vue'
import NpmDonewsTest from 'npm-donews-test'

Vue.use(NpmDonewsTest)

// or
import {
  Banner
  // ...
} from 'npm-donews-test'

Vue.component(Banner.name, Banner)
```
